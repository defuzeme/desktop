00213             QVector<QVariant> columnData;

00215                 columnData << columnStrings[column];
00216 
00217             if (position > indentations.last()) {
00218                 // The last child of the current parent is now the new parent
00219                 // unless the current parent has no children.
00220 
00221                 if (parents.last()->childCount() > 0) {
00222                     parents << parents.last()->child(parents.last()->childCount()-1);

00227                     parents.pop_back();

00232             // Append a new item to the current parent's list of children.
00233             EditableTreeItem *parent = parents.last();
